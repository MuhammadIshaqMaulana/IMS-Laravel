barang_id dan pesanan_id pake integer, karena mereka adalah ID yang berelasi
relasi dibuat di model menggunakan function
   public function pesanan(){ // 1 user hasMany pesanan
        return $this->hasMany('App\Pesanan','user_id', 'id');
    }
   public function user(){ // 1 user hasMany pesanan
        return $this->belongsTo('App\User', 'user_id', 'id');
    }
